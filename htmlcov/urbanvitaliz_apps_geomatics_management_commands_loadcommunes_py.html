3""" 

4Management command to import communes into geomatics 

5 

6 

7This command relies on the csv file with commune / departement / region data 

8available from the following url: 

9 

10https://www.data.gouv.fr/fr/datasets/r/dbe8a621-a9c4-4bc3-9cae-be1699c5ff25 

11 

12

15""" 

16 

17import csv 

18 

19from django.db import transaction 

20 

21from django.core.management.base import BaseCommand 

22 

23from urbanvitaliz.apps.geomatics import models 

24 

25 

26class Command(BaseCommand): 

27 help = "Import France communes/department/region from given CSV file" 

28 

29 def add_arguments(self, parser): 

30 parser.add_argument("filename", type=str) 

31 

32 def handle(self, *args, **options): 

33 filename = options["filename"] 

34 load_communes_from_csv(filename) 

35 

36 

37@transaction.atomic 

38def load_communes_from_csv(filename): 

39 with open(filename) as the_file: 

40 reader = csv.DictReader(the_file) 

41 for row in reader: 

42 load_commune(row) 

43 

44 

45def load_commune(row): 

46 """Create commune department region from row if unknown""" 

47 region = get_region(row["code_region"], row["nom_region"]) 

48 department = get_department(region, row["code_departement"], row["nom_departement"]) 

49 commune, _ = models.Commune.objects.get_or_create( 

50 department=department, 

51 insee=row["code_commune_INSEE"], 

52 postal=row["code_postal"], 

53 defaults={ 

54 "name": row["nom_commune_postal"], 

55 "latitude": row["latitude"] or 0.0, 

56 "longitude": row["longitude"] or 0.0, 

57 }, 

58 ) 

59 return commune 

60 

61 

62def memoize(function): 

63 """Memoize departments and regions to reduce db queries""" 

64 memory = {} 

65 

66 def wrapper(*args, **kwargs): 

67 if args not in memory: 

68 memory[args] = function(*args) 

69 return memory[args] 

70 

71 return wrapper 

72 

73 

74@memoize 

75def get_region(code, name): 

76 region, _ = models.Region.objects.get_or_create(code=code, defaults={"name": name}) 

77 return region 

78 

79 

80@memoize 

81def get_department(region, code, name): 

82 departement, _ = models.Department.objects.get_or_create( 

83 region=region, code=code, defaults={"name": name} 

84 ) 

85 return departement
